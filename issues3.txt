src/components/ProtectedRoute

import React from 'react';
import { Navigate } from 'react-router-dom';
import { useAuth } from '../context/AuthContext';

const ProtectedRoute = ({ children }) => {
  const { user, loading } = useAuth();

  if (loading) {
    return (
      <div style={{ display: 'flex', justifyContent: 'center', alignItems: 'center', minHeight: '100vh' }}>
        <div className="loading neon-text">LOADING...</div>
      </div>
    );
  }

  if (!user) {
    return <Navigate to="/login" />;
  }

  if (!user.isApproved) {
    return <Navigate to="/pending-approval" />;
  }

  return children;
};

export default ProtectedRoute;

pages/games/GameRouter.js
import React, { useState, useEffect } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import api from '../../utils/api';
import CrosswordGame from './CrosswordGame';
import WordScrambleGame from './WordScrambleGame';
import QuickQuizGame from './QuickQuizGame';
import TrueFalseGame from './TrueFalseGame';

const GameRouter = () => {
  const { gameId } = useParams();
  const navigate = useNavigate();
  const [gameType, setGameType] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchGameType();
  }, [gameId]);

  const fetchGameType = async () => {
    try {
      const response = await api.get(`/games/${gameId}`);
      setGameType(response.data.type);
    } catch (error) {
      console.error('Error fetching game type:', error);
      alert('Failed to load game');
      navigate('/games');
    } finally {
      setLoading(false);
    }
  };

  if (loading) {
    return (
      <div style={{ display: 'flex', justifyContent: 'center', alignItems: 'center', minHeight: '80vh' }}>
        <div className="loading neon-text">LOADING...</div>
      </div>
    );
  }

  switch (gameType) {
    case 'crossword':
      return <CrosswordGame />;
    case 'word_scramble':
      return <WordScrambleGame />;
    case 'quick_quiz':
      return <QuickQuizGame />;
    case 'true_false':
      return <TrueFalseGame />;
    default:
      return (
        <div style={{ textAlign: 'center', padding: '40px' }}>
          <h2>Unknown game type</h2>
          <button onClick={() => navigate('/games')} className="retro-btn">
            ← BACK TO GAMES
          </button>
        </div>
      );
  }
};

export default GameRouter;

    '/avatars/avatar8.png',

  ];

    fetchSessions();

  useEffect(() => {
    if (user?.avatar) {
      setSelectedAvatar(user.avatar);

    try {
      const response = await api.get('/auth/sessions');
      setSessions(response.data);
    } catch (error) {
      console.error('Error fetching sessions:', error);
    }
  };

  const handlePasswordChange = async (e) => {
    e.preventDefault();

    if (newPassword !== confirmPassword) {
      alert('New passwords do not match');
      return;
    }

    if (newPassword.length < 6) {
      alert('New password must be at least 6 characters');
      return;
    }

    setChangingPassword(true);

    try {
      await api.put('/auth/change-password', {
        currentPassword,
        newPassword
      });

      alert('Password changed successfully!');
      setCurrentPassword('');
      setNewPassword('');
      setConfirmPassword('');
    } catch (error) {
      alert(error.response?.data?.message || 'Failed to change password');
    } finally {
      setChangingPassword(false);
    }

  const handleAvatarChange = async (avatarUrl) => {
    try {
      const response = await api.put('/auth/update-profile', {
        avatar: avatarUrl

import React, { useState, useEffect } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import { motion, AnimatePresence } from 'framer-motion';
import { useAuth } from '../context/AuthContext';
import api from '../utils/api';
import { getImageUrl } from '../utils/getImageUrl';
import LevelUpToast from '../components/LevelUpToast';

const TopicDetail = () => {
  const { topicId } = useParams();
  const navigate = useNavigate();
  const { user, updateUser } = useAuth();

  const [topic, setTopic] = useState(null);
  const [loading, setLoading] = useState(true);
  const [currentStage, setCurrentStage] = useState('materials'); // materials, mandatory, bonus, completed
  const [materialsConfirmed, setMaterialsConfirmed] = useState(false);

  // Mandatory question state
  const [mandatoryQuestion, setMandatoryQuestion] = useState(null);
  const [selectedAnswer, setSelectedAnswer] = useState(null);
  const [submitting, setSubmitting] = useState(false);
  const [mandatoryResult, setMandatoryResult] = useState(null);

  // Bonus questions state
  const [bonusQuestions, setBonusQuestions] = useState([]);
  const [currentBonusIndex, setCurrentBonusIndex] = useState(0);
  const [bonusAnswers, setBonusAnswers] = useState({});
  const [bonusResults, setBonusResults] = useState({});
  const [bonusScore, setBonusScore] = useState(0);

  // Completion state
  const [topicCompleted, setTopicCompleted] = useState(false);
  const [nextTopic, setNextTopic] = useState(null);

  // Level up toast
  const [showLevelUp, setShowLevelUp] = useState(false);
  const [newLevel, setNewLevel] = useState(null);

  useEffect(() => {
    fetchTopic();
  }, [topicId]);

  const fetchTopic = async () => {
    try {
      const response = await api.get(`/topics/${topicId}`);
      setTopic(response.data);

      // Check if already completed
      if (response.data.userProgress?.completed) {
        setTopicCompleted(true);
        setCurrentStage('completed');
      } else if (response.data.userProgress?.mandatoryPassed) {
        // Mandatory passed, show bonus questions
        setCurrentStage('bonus');
        loadBonusQuestions(response.data);
      } else {
        // Start from materials
        setCurrentStage('materials');
      }
    } catch (error) {
      console.error('Error fetching topic:', error);
      alert('Failed to load topic');
      navigate('/topics');
    } finally {
      setLoading(false);
    }
  };

  const loadBonusQuestions = (topicData) => {
    const bonus = topicData.questions.filter(q => !q.isMandatory);
    setBonusQuestions(bonus);

    // Load previously answered bonus questions
    if (topicData.userProgress?.bonusAnswers) {
      setBonusAnswers(topicData.userProgress.bonusAnswers);
      
      // Calculate score from previous answers
      let score = 0;
      Object.entries(topicData.userProgress.bonusAnswers).forEach(([qId, answer]) => {
        const question = bonus.find(q => q._id === qId);
        if (question && answer === question.correctAnswer) {
          score += 50;
        }
      });
      setBonusScore(score);
    }
  };

  const handleConfirmMaterials = () => {
    setMaterialsConfirmed(true);
    setCurrentStage('mandatory');
    
    // Load mandatory question
    const mandatory = topic.questions.find(q => q.isMandatory);
    setMandatoryQuestion(mandatory);
  };

  const handleSubmitMandatory = async (e) => {
    e.preventDefault();
    
    if (selectedAnswer === null) {
      alert('Please select an answer');
      return;
    }

    setSubmitting(true);

    try {
      const oldLevel = user.level; // Store old level for comparison

      const response = await api.post(`/topics/${topicId}/submit-mandatory`, {
        answer: selectedAnswer
      });

      if (response.data.passed) {
        setMandatoryResult({
          passed: true,
          correctAnswer: true,
          badgeImage: response.data.badgeImage
        });

        // Check for level up
        if (response.data.leveledUp || (response.data.newLevel && response.data.newLevel > oldLevel)) {
          setNewLevel(response.data.newLevel);
          setShowLevelUp(true);
        }

        // Refresh user data
        const userResponse = await api.get('/auth/me');
        updateUser(userResponse.data);

        // Auto-advance to bonus after 3 seconds
        setTimeout(() => {
          setCurrentStage('bonus');
          loadBonusQuestions(topic);
        }, 3000);
      } else {
        // Wrong answer
        setMandatoryResult({
          passed: false,
          correctAnswer: false,
          correctAnswerIndex: response.data.correctAnswerIndex,
          explanation: mandatoryQuestion.explanation
        });

        // Reset after showing error
        setTimeout(() => {
          setMandatoryResult(null);
          setSelectedAnswer(null);
        }, 5000);
      }
    } catch (error) {
      console.error('Error submitting mandatory:', error);
      alert('Error submitting answer');
    } finally {
      setSubmitting(false);
    }
  };

  const handleSubmitBonus = async (questionId, answer) => {
    try {
      const oldLevel = user.level;

      const response = await api.post(`/topics/${topicId}/submit-bonus`, {
        questionId,
        answer
      });

      // Update local state
      setBonusAnswers(prev => ({ ...prev, [questionId]: answer }));
      setBonusResults(prev => ({
        ...prev,
        [questionId]: {
          correct: response.data.correct,
          correctAnswer: response.data.correctAnswer
        }
      }));

      if (response.data.correct) {
        setBonusScore(prev => prev + 50);
      }

      // Check for level up
      if (response.data.leveledUp || (response.data.newLevel && response.data.newLevel > oldLevel)) {
        setNewLevel(response.data.newLevel);
        setShowLevelUp(true);
      }

      // Refresh user data
      const userResponse = await api.get('/auth/me');
      updateUser(userResponse.data);

      // Move to next question or complete
      if (currentBonusIndex < bonusQuestions.length - 1) {
        setTimeout(() => {
          setCurrentBonusIndex(currentBonusIndex + 1);
        }, 2000);
      } else {
        // All bonus questions done
        setTimeout(() => {
          setTopicCompleted(true);
          setCurrentStage('completed');
          fetchNextTopic();
        }, 2000);
      }
    } catch (error) {
      console.error('Error submitting bonus:', error);
      alert('Error submitting answer');
    }
  };

  const fetchNextTopic = async () => {
    try {
      const response = await api.get('/topics/recommended-next');
      setNextTopic(response.data);
    } catch (error) {
      console.error('Error fetching next topic:', error);
    }
  };

  const handleSkipBonus = () => {
    setTopicCompleted(true);
    setCurrentStage('completed');
    fetchNextTopic();
  };

const filterTopics = () => {
  let filtered = [...topics];

  // Search filter
  if (searchTerm) {
    filtered = filtered.filter(topic =>
      topic.title.toLowerCase().includes(searchTerm.toLowerCase()) ||
      topic.description.toLowerCase().includes(searchTerm.toLowerCase())
    );
  }

  // Status filter
  if (statusFilter !== 'all') {
    filtered = filtered.filter(topic => topic.status === statusFilter);
  }

  // Sort
  switch (sortBy) {
    case 'alphabetical':
      filtered.sort((a, b) => a.title.localeCompare(b.title));
      break;
    case 'newest':
      filtered.sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));
      break;
    case 'status':
      // Sort by status priority: not_started > in_progress > completed
      const statusPriority = { 'not_started': 0, 'in_progress': 1, 'completed': 2 };
      filtered.sort((a, b) => statusPriority[a.status] - statusPriority[b.status]);
      break;
    default:
      break;
  }

  setFilteredTopics(filtered);
};

import { motion } from 'framer-motion';
import api from '../../utils/api';

const UserManagement = () => {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [searchTerm, setSearchTerm] = useState('');
  const [filterRole, setFilterRole] = useState('all');
  const [filterStatus, setFilterStatus] = useState('all');

  useEffect(() => {
    fetchUsers();
  }, []);

  const fetchUsers = async () => {
    try {
      const response = await api.get('/admin/users');
      setUsers(response.data);
    } catch (error) {
      console.error('Error fetching users:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleApprove = async (userId) => {
    if (!window.confirm('Approve this user?')) return;

    try {
      await api.post(`/admin/users/${userId}/approve`);
      alert('User approved successfully!');
      fetchUsers();
    } catch (error) {
      alert('Failed to approve user');
    }
  };

  const handleDelete = async (userId) => {
    if (!window.confirm('Delete this user? This action cannot be undone!')) return;

    try {
      await api.delete(`/admin/users/${userId}`);
      alert('User deleted successfully');
      fetchUsers();
    } catch (error) {
      alert('Failed to delete user');
    }
  };

  const filteredUsers = users.filter(user => {
    const matchesSearch = user.username.toLowerCase().includes(searchTerm.toLowerCase()) ||
                         user.email.toLowerCase().includes(searchTerm.toLowerCase());
    const matchesRole = filterRole === 'all' || user.role === filterRole;
    const matchesStatus = filterStatus === 'all' || 
                         (filterStatus === 'approved' && user.isApproved) ||
                         (filterStatus === 'pending' && !user.isApproved);
    
    return matchesSearch && matchesRole && matchesStatus;
  });

  if (loading) {
    return (
      <div style={{ display: 'flex', justifyContent: 'center', alignItems: 'center', minHeight: '80vh' }}>
        <div className="loading neon-text">LOADING...</div>

      </h1>

        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        className="retro-card"
        style={{ marginBottom: '30px' }}
      >
        <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit, minmax(200px, 1fr))', gap: '15px' }}>
          <div>
            <label style={{ display: 'block', marginBottom: '8px', fontSize: '10px', color: 'var(--text-medium)' }}>
              SEARCH
            </label>
            <input
              type="text"
              className="retro-input"
              placeholder="Username or email..."
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
            />
          </div>

          <div>
            <label style={{ display: 'block', marginBottom: '8px', fontSize: '10px', color: 'var(--text-medium)' }}>
              ROLE
            </label>
            <select
              className="retro-input"
              value={filterRole}
              onChange={(e) => setFilterRole(e.target.value)}
            >
              <option value="all">All Roles</option>
              <option value="admin">Admin</option>
              <option value="employee">Employee</option>
            </select>
          </div>

          <div>
            <label style={{ display: 'block', marginBottom: '8px', fontSize: '10px', color: 'var(--text-medium)' }}>
              STATUS
            </label>
            <select
              className="retro-input"
              value={filterStatus}

                <div style={{
                  padding: '8px 15px',
                  background: 'var(--success-green)',
                  color: 'white',
                  fontSize: '11px',
                  fontWeight: 'bold',
                  borderRadius: '5px'
                }}>
                  {topic.completionRate}%
                </div>
              </div>
            ))}
          </div>
        </motion.div>
      )}

      {/* Top Users */}
      {analytics?.topUsers && analytics.topUsers.length > 0 && (
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ delay: 0.5 }}
          className="retro-card"
        >
          <h3 style={{ fontSize: '14px', color: 'var(--secondary-pink)', marginBottom: '20px' }}>
            🏆 TOP LEARNERS
          </h3>

          <div style={{ display: 'flex', flexDirection: 'column', gap: '10px' }}>
            {analytics.topUsers.map((user, index) => (
              <div
                key={user._id}
                style={{
                  padding: '15px',
                  background: 'var(--bg-light)',
                  border: '2px solid var(--border-color)',
                  display: 'flex',
                  justifyContent: 'space-between',
                  alignItems: 'center'
                }}
              >
                <div>
                  <div style={{ fontSize: '12px', fontWeight: 'bold', marginBottom: '5px' }}>
                    {index + 1}. {user.username}
                  </div>
                  <div style={{ fontSize: '10px', color: 'var(--text-medium)' }}>
                    Level {user.level} • {user.badges} badges • {user.completedTopics} topics
                  </div>
                </div>
                <div style={{
                  padding: '8px 15px',
                  background: 'var(--bright-blue)',
                  color: 'white',
                  fontSize: '11px',
                  fontWeight: 'bold',
                  borderRadius: '5px'
                }}>
                  {user.xp.toLocaleString()} XP
                </div>
              </div>
            ))}
          </div>
        </motion.div>
      )}
    </div>
  );
};

// Get admin stats
router.get('/stats', authenticateToken, isAdmin, async (req, res) => {
  try {
    const totalUsers = await User.countDocuments();
    const pendingApprovals = await User.countDocuments({ isApproved: false, role: 'employee' });
    const activeTopics = await Topic.countDocuments({ isActive: true });
    const totalTopics = await Topic.countDocuments();
    
    const allUsers = await User.find({});
    const totalCompletions = allUsers.reduce((sum, user) => sum + (user.completedTopics?.length || 0), 0);
    const totalBadges = allUsers.reduce((sum, user) => sum + (user.badges?.length || 0), 0);

    res.json({
      totalUsers,
      pendingApprovals,
      activeTopics,
      totalTopics,
      totalCompletions,
      totalBadges
    });
  } catch (error) {
    console.error('Error fetching admin stats:', error);
    res.status(500).json({ message: 'Server error' });
  }
});

// Get analytics
router.get('/analytics', authenticateToken, isAdmin, async (req, res) => {
  try {
    const users = await User.find({ role: 'employee' });
    
    const avgXpPerUser = users.reduce((sum, u) => sum + u.xp, 0) / users.length || 0;
    const avgLevel = users.reduce((sum, u) => sum + u.level, 0) / users.length || 0;
    const totalBadges = users.reduce((sum, u) => sum + (u.badges?.length || 0), 0);
    
    const topUsers = users
      .sort((a, b) => b.xp - a.xp)
      .slice(0, 10)
      .map(u => ({
        _id: u._id,
        username: u.username,
        level: u.level,
        xp: u.xp,
        badges: u.badges?.length || 0,
        completedTopics: u.completedTopics?.length || 0
      }));

    const topics = await Topic.find({});
    const topTopics = topics.map(t => {
      const completions = users.filter(u => 
        u.completedTopics?.some(ct => ct.topicId.toString() === t._id.toString())
      ).length;
      return {
        title: t.title,
        completions,
        completionRate: Math.round((completions / users.length) * 100)
      };
    }).sort((a, b) => b.completions - a.completions).slice(0, 10);

    res.json({
      avgXpPerUser: Math.round(avgXpPerUser),
      avgLevel: Math.round(avgLevel),
      totalBadges,
      avgCompletionRate: Math.round(topTopics.reduce((sum, t) => sum + t.completionRate, 0) / topTopics.length || 0),
      topUsers,
      topTopics
    });
  } catch (error) {
    console.error('Error fetching analytics:', error);
    res.status(500).json({ message: 'Server error' });
  }
});
