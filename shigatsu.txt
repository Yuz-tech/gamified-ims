// Updated fetchNextTopic function
const fetchNextTopic = async () => {
  try {
    // Fetch all topics
    const topicsResponse = await api.get('/topics');
    const allTopics = topicsResponse.data;

    // Get fresh user data to ensure we have latest completions
    const userResponse = await api.get('/auth/me');
    const currentUser = userResponse.data;

    // Find first incomplete topic (not in user's completedTopics) that's not current topic
    const incompleteTopic = allTopics.find(t => {
      const isCompleted = currentUser.completedTopics?.some(ct => ct.topicId === t._id);
      const isCurrentTopic = t._id === topicId;
      const isActive = t.isActive;
      
      return !isCompleted && !isCurrentTopic && isActive;
    });

    if (incompleteTopic) {
      setNextTopic(incompleteTopic);
      console.log('✅ Next incomplete topic:', incompleteTopic.title);
    } else {
      setNextTopic(null);
      console.log('🎉 All topics completed!');
    }
  } catch (error) {
    console.error('Error fetching next topic:', error);
  }
};

javascriptconst getFilteredTopics = () => {
  let filtered = topics;

  // Search filter
  if (searchTerm) {
    filtered = filtered.filter(topic =>
      topic.title.toLowerCase().includes(searchTerm.toLowerCase()) ||
      topic.description.toLowerCase().includes(searchTerm.toLowerCase())
    );
  }

  // Status filter
  if (filterStatus !== 'all') {
    filtered = filtered.filter(topic => {
      const isCompleted = user?.completedTopics?.some(ct => ct.topicId === topic._id);
      
      if (filterStatus === 'completed') return isCompleted;
      if (filterStatus === 'not_started') return !isCompleted;
      return true;
    });
  }

  // ✅ SIMPLE SORT - Backend already sorted, just maintain order
  return filtered.sort((a, b) => {
    // Sort by priority first
    const priorityDiff = (a.priority || 999) - (b.priority || 999);
    if (priorityDiff !== 0) return priorityDiff;
    
    // Then alphabetically
    return a.title.localeCompare(b.title);
  });
};
