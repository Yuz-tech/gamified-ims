import mongoose from 'mongoose';

const gameCompletionSchema = new mongoose.Schema({
  userId: {
    type: mongoose.Schema.Types.ObjectId,
    ref: 'User',
    required: true
  },
  score: {
    type: Number,
    required: true
  },
  timeSpent: {
    type: Number,
    required: true
  },
  completedAt: {
    type: Date,
    default: Date.now
  }
});

const gameSchema = new mongoose.Schema({
  title: {
    type: String,
    required: true
  },
  description: {
    type: String,
    required: true
  },
  gameType: {
    type: String,
    required: true,
    enum: ['texttwist', 'wordle', 'quickquiz', 'hangman']
  },
  difficulty: {
    type: String,
    enum: ['easy', 'medium', 'hard'],
    default: 'medium'
  },
  maxXP: {
    type: Number,
    default: 200
  },
  timeLimit: {
    type: Number,
    default: 0
  },
  content: {
    type: mongoose.Schema.Types.Mixed,
    required: true
  },
  isActive: {
    type: Boolean,
    default: true
  },
  // Track who has completed this game
  completions: [gameCompletionSchema]
}, {
  timestamps: true
});

// Method to check if user has completed this game
gameSchema.methods.hasUserCompleted = function(userId) {
  return this.completions.some(
    completion => completion.userId.toString() === userId.toString()
  );
};

// Method to add completion
gameSchema.methods.addCompletion = function(userId, score, timeSpent) {
  this.completions.push({
    userId,
    score,
    timeSpent
  });
  return this.save();
};

export default mongoose.model('Game', gameSchema);

import express from 'express';
import Game from '../models/Game.js';
import User from '../models/User.js';
import { authenticateToken, isAdmin } from '../middleware/auth.js';
import { logActivity } from '../utils/activityLogger.js';

const router = express.Router();

// ==================== USER ROUTES ====================

// Get all active games with completion status
router.get('/', authenticateToken, async (req, res) => {
  try {
    const games = await Game.find({ isActive: true })
      .select('title description gameType difficulty maxXP timeLimit')
      .sort({ createdAt: -1 });

    // Add completion status for current user
    const gamesWithStatus = games.map(game => {
      const hasCompleted = game.hasUserCompleted(req.user._id);
      return {
        ...game.toObject(),
        hasCompleted,
        canPlay: !hasCompleted
      };
    });

    res.json(gamesWithStatus);
  } catch (error) {
    console.error('Error fetching games:', error);
    res.status(500).json({ message: 'Server error' });
  }
});

// Get single game by ID (with content)
router.get('/play/:id', authenticateToken, async (req, res) => {
  try {
    const gameId = req.params.id;
    const game = await Game.findById(gameId);

    if (!game) {
      return res.status(404).json({ message: 'Game not found' });
    }

    if (!game.isActive) {
      return res.status(403).json({ message: 'This game is not active' });
    }

    // Check if user has already completed this game
    const hasCompleted = game.hasUserCompleted(req.user._id);
    
    if (hasCompleted) {
      return res.status(403).json({ 
        message: 'You have already completed this game and earned XP from it',
        hasCompleted: true
      });
    }

    res.json({
      ...game.toObject(),
      hasCompleted: false,
      canPlay: true
    });
  } catch (error) {
    console.error('Error fetching game:', error);
    res.status(500).json({ message: 'Server error', error: error.message });
  }
});

// Submit game score (only once per game per user)
router.post('/submit-score', authenticateToken, async (req, res) => {
  try {
    const { gameId, gameType, score, timeSpent } = req.body;
    const userId = req.user._id;

    if (typeof score !== 'number' || score < 0) {
      return res.status(400).json({ message: 'Invalid score' });
    }

    // Get game
    const game = await Game.findById(gameId);
    if (!game) {
      return res.status(404).json({ message: 'Game not found' });
    }

    // Check if user has already completed this game
    if (game.hasUserCompleted(userId)) {
      return res.status(403).json({ 
        message: 'You have already completed this game',
        alreadyCompleted: true
      });
    }

    const maxXP = game.maxXP;
    const cappedScore = Math.min(score, maxXP);

    // Award XP to user
    const user = await User.findById(userId);
    if (!user) {
      return res.status(404).json({ message: 'User not found' });
    }

    const oldLevel = user.level;
    user.xp += cappedScore;

    // Calculate new level
    const { calculateLevel } = await import('../utils/levelSystem.js');
    const newLevel = calculateLevel(user.xp);
    const leveledUp = newLevel > oldLevel;

    if (leveledUp) {
      user.level = newLevel;
    }

    await user.save();

    // Mark game as completed for this user
    await game.addCompletion(userId, cappedScore, timeSpent);

    // Log activity
    await logActivity(userId, 'game_completed', {
      gameId: game._id,
      gameType,
      score: cappedScore,
      timeSpent,
      leveledUp
    }, req);

    res.json({
      xpEarned: cappedScore,
      totalXP: user.xp,
      leveledUp,
      newLevel: user.level,
      completed: true
    });
  } catch (error) {
    console.error('Error submitting game score:', error);
    res.status(500).json({ message: 'Server error' });
  }
});

// Update game (admin)
router.put('/admin/:id', authenticateToken, isAdmin, async (req, res) => {
  try {
    const gameId = req.params.id;
    const { title, description, difficulty, maxXP, timeLimit, content, isActive } = req.body;

    const game = await Game.findByIdAndUpdate(
      gameId,
      {
        title,
        description,
        difficulty,
        maxXP,
        timeLimit,
        content,
        isActive
      },
      { new: true }
    );

    if (!game) {
      return res.status(404).json({ message: 'Game not found' });
    }

    await logActivity(req.user._id, 'game_updated', {
      gameId: game._id,
      title: game.title
    }, req);

    res.json({
      message: 'Game updated successfully',
      game
    });
  } catch (error) {
    console.error('Error updating game:', error);
    res.status(500).json({ message: 'Server error' });
  }
});

// Delete game (admin)
router.delete('/admin/:id', authenticateToken, isAdmin, async (req, res) => {
  try {
    const gameId = req.params.id;

    const game = await Game.findByIdAndDelete(gameId);

    if (!game) {

    }

    await logActivity(req.user._id, 'game_deleted', {

    res.json({ message: 'Game deleted successfully' });

import User from '../models/User.js';
import Badge from '../models/Badge.js';

/**
 * Award badge to user
 * - If user already has badge for this topic: increment badgeCount
 * - If user doesn't have badge: create new entry with badgeCount: 1
 * - Also updates Badge model's badgeCount for current year
 */
async function awardBadgeToUser(userId, topicId, badgeName, badgeImage) {
  try {
    const user = await User.findById(userId);
    if (!user) {
      throw new Error('User not found');
    }

    // Check if user already has this badge (from a previous year)
    const existingBadgeIndex = user.badges.findIndex(
      badge => badge.topicId.toString() === topicId.toString()
    );

    if (existingBadgeIndex !== -1) {
      // ✅ User already has this badge - INCREMENT badgeCount
      user.badges[existingBadgeIndex].badgeCount += 1;
      user.badges[existingBadgeIndex].earnedAt = new Date(); // Update earned date
      
      console.log(`✨ User ${user.username} re-earned badge for topic ${topicId}. New count: ${user.badges[existingBadgeIndex].badgeCount}`);
    } else {
      // ✅ First time earning this badge - ADD new entry
      user.badges.push({
        topicId,
        badgeName,
        badgeImage,
        earnedAt: new Date(),
        badgeCount: 1
      });
      
      console.log(`🏆 User ${user.username} earned badge for topic ${topicId} for the first time`);
    }

    await user.save();

    // ✅ Also increment the Badge model's badgeCount
    // This tracks total instances of this badge earned across all users
    const currentYear = new Date().getFullYear();
    const badge = await Badge.findOne({ topicId, year: currentYear });
    
    if (badge) {
      badge.badgeCount += 1;
      await badge.save();
      console.log(`📊 Badge ${badgeName} total count for ${currentYear}: ${badge.badgeCount}`);
    } else {
      // Create badge record for current year if it doesn't exist
      await Badge.create({
        name: badgeName,
        description: `Badge earned by completing topic`,
        imageUrl: badgeImage,
        topicId,
        year: currentYear,
        badgeCount: 1,
        isActive: true
      });
      console.log(`🆕 Created badge record for ${badgeName} in year ${currentYear}`);
    }

    return {
      success: true,
      isReEarned: existingBadgeIndex !== -1,
      badgeCount: existingBadgeIndex !== -1 
        ? user.badges[existingBadgeIndex].badgeCount 
        : 1
    };
  } catch (error) {
    console.error('Error awarding badge:', error);
    throw error;
  }
}

/**
 * Get user's badge count for a specific topic
 */
async function getUserBadgeCount(userId, topicId) {
  try {
    const user = await User.findById(userId);
    if (!user) return 0;

    const badge = user.badges.find(
      b => b.topicId.toString() === topicId.toString()
    );

    return badge ? badge.badgeCount : 0;
  } catch (error) {
    console.error('Error getting user badge count:', error);
    return 0;
  }
}

/**
 * Get total badge count across all users for a topic in a specific year
 */
async function getTotalBadgeCount(topicId, year = new Date().getFullYear()) {
  try {
    const badge = await Badge.findOne({ topicId, year });
    return badge ? badge.badgeCount : 0;
  } catch (error) {
    console.error('Error getting total badge count:', error);
    return 0;
  }
}

  getTotalBadgeCount

router.post('/:topicId/submit-answers', authenticateToken, async (req, res) => {
  try {
    const { topicId } = req.params;
    const { answers } = req.body;
    const userId = req.user._id;

    // Get topic
    const topic = await Topic.findById(topicId);
    if (!topic) {
      return res.status(404).json({ message: 'Topic not found' });
    }

    // Get user
    const user = await User.findById(userId);
    if (!user) {
      return res.status(404).json({ message: 'User not found' });
    }

    // Check answers
    let correctCount = 0;
    let mandatoryCorrect = false;
    let bonusCorrect = 0;

    answers.forEach((answer, index) => {
      const question = topic.questions[index];
      if (question && answer === question.correctAnswer) {
        correctCount++;
        if (question.isMandatory) {
          mandatoryCorrect = true;
        } else {
          bonusCorrect++;
        }
      }
    });

    const mandatoryCompleted = mandatoryCorrect;
    const bonusCompleted = bonusCorrect === 4; // All 4 bonus questions correct

    // Calculate XP
    let xpEarned = 0;
    if (mandatoryCompleted) xpEarned += 50;
    if (bonusCompleted) xpEarned += 50;

    const oldLevel = user.level;
    user.xp += xpEarned;

    // Calculate new level
    const { calculateLevel } = await import('../utils/levelSystem.js');
    const newLevel = calculateLevel(user.xp);
    const leveledUp = newLevel > oldLevel;
    if (leveledUp) {
      user.level = newLevel;
    }

    // Update or add completion record
    const existingCompletion = user.completedTopics.find(
      ct => ct.topicId.toString() === topicId.toString()
    );

    if (existingCompletion) {
      existingCompletion.mandatoryCompleted = mandatoryCompleted;
      existingCompletion.bonusCompleted = bonusCompleted;
      existingCompletion.bonusCorrect = bonusCorrect;
      existingCompletion.completedAt = new Date();
    } else {
      user.completedTopics.push({
        topicId,
        mandatoryCompleted,
        bonusCompleted,
        bonusCorrect,
        completedAt: new Date()
      });
    }

    await user.save();

    // ✅ AWARD BADGE using new logic (if mandatory completed)
    let badgeInfo = null;
    if (mandatoryCompleted) {
      const badgeResult = await awardBadgeToUser(
        userId,
        topicId,
        topic.badgeName,
        topic.badgeImage
      );
      
      badgeInfo = {
        earned: true,
        isReEarned: badgeResult.isReEarned,
        badgeCount: badgeResult.badgeCount,
        badgeName: topic.badgeName,
        badgeImage: topic.badgeImage
      };
    }

    // Log activity
    await logActivity(userId, 'topic_completed', {
      topicId,
      topicTitle: topic.title,
      score: correctCount,
      mandatoryCompleted,
      bonusCompleted,
      xpEarned,
      leveledUp,
      badgeEarned: mandatoryCompleted,
      badgeReEarned: badgeInfo?.isReEarned,
      badgeCount: badgeInfo?.badgeCount
    }, req);

    res.json({
      correctCount,
      totalQuestions: topic.questions.length,
      mandatoryCompleted,
      bonusCompleted,
      bonusCorrect,
      xpEarned,
      totalXP: user.xp,
      leveledUp,
      newLevel: user.level,
      badge: badgeInfo
    });
  } catch (error) {
    console.error('Error submitting answers:', error);
    res.status(500).json({ message: 'Server error' });
  }
});
