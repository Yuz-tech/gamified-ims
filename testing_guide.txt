javascript// Test levelSystem calculations on frontend
describe('Frontend Level Calculations', () => {
  const levelThresholds = [
    0, 100, 250, 450, 700, 1000, 1350, 1750, 2200, 2700,
    3250, 3850, 4500, 5200, 5950, 6750, 7600, 8500, 9450, 10450,
    11500, 12600, 13750, 14950, 16200, 17500, 18850, 20250, 21700, 23200
  ];

  const getXPForLevel = (level) => {
    if (level <= 30) {
      return levelThresholds[level - 1] || 0;
    }
    const levelsAbove30 = level - 30;
    const baseXP = 23200;
    const linearGrowth = levelsAbove30 * 1500;
    const exponentialGrowth = (levelsAbove30 * (levelsAbove30 + 1) / 2) * 50;
    return baseXP + linearGrowth + exponentialGrowth;
  };

  test('should calculate XP for level 1', () => {
    expect(getXPForLevel(1)).toBe(0);
  });

  test('should calculate XP for level 15', () => {
    expect(getXPForLevel(15)).toBe(5950);
  });

  test('should calculate XP for level 30', () => {
    expect(getXPForLevel(30)).toBe(23200);
  });

  test('should calculate progress correctly', () => {
    const userXP = 150;
    const userLevel = 2;

    const currentLevelXP = getXPForLevel(userLevel);
    const nextLevelXP = getXPForLevel(userLevel + 1);
    const progress = ((userXP - currentLevelXP) / (nextLevelXP - currentLevelXP)) * 100;

    expect(progress).toBeGreaterThan(0);
    expect(progress).toBeLessThan(100);
  });
});

javascriptimport request from 'supertest';
import mongoose from 'mongoose';
import { MongoMemoryServer } from 'mongodb-memory-server';
import app from '../../server.js';

let mongoServer;
let authToken;

beforeAll(async () => {
  mongoServer = await MongoMemoryServer.create();
  await mongoose.connect(mongoServer.getUri());
});

afterAll(async () => {
  await mongoose.disconnect();
  await mongoServer.stop();
});

describe('Complete User Flow: Topic → Game → Leaderboard', () => {
  test('should complete entire flow successfully', async () => {
    // 1. Register
    const registerRes = await request(app)
      .post('/api/auth/register')
      .send({
        username: 'flowtest',
        email: 'flow@test.com',
        password: 'Test123!'
      });

    expect(registerRes.status).toBe(201);
    authToken = registerRes.body.token;
    const userId = registerRes.body.user._id;

    // 2. Get topics
    const topicsRes = await request(app)
      .get('/api/topics')
      .set('Authorization', `Bearer ${authToken}`);

    expect(topicsRes.status).toBe(200);
    const topicId = topicsRes.body[0]._id;

    // 3. Complete topic mandatory
    const mandatoryRes = await request(app)
      .post(`/api/topics/${topicId}/submit-mandatory`)
      .set('Authorization', `Bearer ${authToken}`)
      .send({ answer: topicsRes.body[0].questions[0].correctAnswer });

    expect(mandatoryRes.status).toBe(200);
    expect(mandatoryRes.body.xpEarned).toBe(100);

    // 4. Play game
    const gamesRes = await request(app)
      .get('/api/games')
      .set('Authorization', `Bearer ${authToken}`);

    const gameId = gamesRes.body[0]._id;

    const gameSubmitRes = await request(app)
      .post(`/api/games/${gameId}/submit`)
      .set('Authorization', `Bearer ${authToken}`)
      .send({
        answers: {
          0: gamesRes.body[0].questions[0].answer,
          1: gamesRes.body[0].questions[1].answer
        }
      });

    expect(gameSubmitRes.status).toBe(200);

    // 5. Check leaderboard
    const leaderboardRes = await request(app)
      .get('/api/leaderboard')
      .set('Authorization', `Bearer ${authToken}`);

    expect(leaderboardRes.status).toBe(200);
    const userInLeaderboard = leaderboardRes.body.find(u => u._id === userId);
    expect(userInLeaderboard).toBeDefined();
    expect(userInLeaderboard.xp).toBeGreaterThan(0);
  });
});

javascriptimport http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
  stages: [
    { duration: '30s', target: 10 }, // Ramp up to 10 users
    { duration: '1m', target: 10 },  // Stay at 10 users
    { duration: '30s', target: 20 }, // Ramp up to 20 users
    { duration: '1m', target: 20 },  // Stay at 20 users
    { duration: '30s', target: 0 },  // Ramp down to 0
  ],
  thresholds: {
    http_req_duration: ['p(95)<500'], // 95% of requests must complete within 500ms
    http_req_failed: ['rate<0.1'],    // Error rate must be below 10%
  },
};

const BASE_URL = 'http://localhost:5000/api';

export default function () {
  // Register
  let registerRes = http.post(`${BASE_URL}/auth/register`, JSON.stringify({
    username: `user_${__VU}_${Date.now()}`,
    email: `user_${__VU}_${Date.now()}@test.com`,
    password: 'Test123!'
  }), {
    headers: { 'Content-Type': 'application/json' }
  });

  check(registerRes, {
    'registration successful': (r) => r.status === 201,
    'token received': (r) => r.json('token') !== undefined,
  });

  const token = registerRes.json('token');

  // Get topics
  let topicsRes = http.get(`${BASE_URL}/topics`, {
    headers: { 'Authorization': `Bearer ${token}` }
  });

  check(topicsRes, {
    'topics loaded': (r) => r.status === 200,
    'topics array': (r) => Array.isArray(r.json()),
  });

  // Get leaderboard
  let leaderboardRes = http.get(`${BASE_URL}/leaderboard`, {
    headers: { 'Authorization': `Bearer ${token}` }
  });

  check(leaderboardRes, {
    'leaderboard loaded': (r) => r.status === 200,
  });

  sleep(1);
}
